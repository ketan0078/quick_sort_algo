#include<stdio.h>


int partition(int A[], int l, int h){
	int pivot = A[l];
	int i,j,temp; i=l; j=h;
	while(i<j){
		while(A[i]<=pivot)
			i++;
		while(A[j]>pivot)
			j--;
		if (i<j)
		{
			temp= A[i];
			A[i]=A[j];
			A[j]=temp;
		}
	}
	temp=A[l];
	A[l]=A[j];
	A[j]=temp;

	return j;
}
void quick_sort(int A[], int l , int h){
	
	if (l<h)
	{
		j= partition(A,l,h);
		quick_sort(A,l,j);
		quick_sort(A,j+1,h);
	}
}
int main(){
	int A[]= {17,13,14,46,23,37,43,12,1,39};
	quick_sort(A,0,9);
	for (int i = 0; i <10;i++)
	{
	   std::cout<<A[i]<<" ";
	}
	return 0;
}
